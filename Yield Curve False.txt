
Public Sub YCFalse()
    
    Dim rsSheets As ADODB.Recordset
    Dim WB As Workbook
    Dim WsX As Worksheet
    Dim i As Integer
    Dim rowName As Integer
    Dim newName As String
    Dim NewFile As String
    Dim GNArr As Variant
    Dim sheetNama As String

    
    GNArr = Array("Gross", "Net")
    
    Set WB = ThisWorkbook
    countRow = 0
    For Each GN In GNArr
        'Populating data
        Set rsSheets = AddSheetsGross
        rsSheets.MoveFirst
        
        If Not rsSheets.EOF Then
            Do While Not rsSheets.EOF
                If GN = "Gross" Then
                    If rsSheets.Fields("Sheet Name").Value = "Auto" Then
                        sheetNama = rsSheets.Fields("Sheet Name").Value & "_Gross"
                    Else
                        sheetNama = rsSheets.Fields("Sheet Name").Value & "_Grs"
                    End If
                Else
                    sheetNama = rsSheets.Fields("Sheet Name").Value & "_" & GN
                End If
                
                Set WsX = WB.Sheets(sheetNama)
                
                WsX.Cells(399, 1).ClearContents
                WsX.Cells(407, 1).ClearContents
                
                For i = 1 To 23
                    
                    WsX.Cells(399, 2 + i).ClearContents
                    WsX.Cells(397, 2 + i).FormulaR1C1 = "=R396C" & (2 + i) & " * (1+ 'Discounting pattern'!DiscRateStat)^-(R394C" & (2 + i) & ")"
                    
                    WsX.Cells(407, 2 + i).ClearContents
                    WsX.Cells(405, 2 + i).FormulaR1C1 = "=R404C" & (2 + i) & " * (1+ 'Discounting pattern'!DiscRateStat)^-(R402C" & (2 + i) & ")"
                    
                
                Next i
                
                rsSheets.MoveNext
            Loop
        End If
        
    Next GN
    

    WB.Close SaveChanges:=False
    
    MsgBox "End"

End Sub
